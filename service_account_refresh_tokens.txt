-- V2__Service_Account_Refresh_Tokens.sql
-- Add refresh tokens table for service accounts

CREATE TABLE IF NOT EXISTS service_account_refresh_tokens (
    id BIGSERIAL PRIMARY KEY,
    token VARCHAR(500) UNIQUE NOT NULL,
    service_account_id BIGINT NOT NULL,
    expiry_date TIMESTAMP NOT NULL,
    revoked BOOLEAN NOT NULL DEFAULT false,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (service_account_id) REFERENCES service_accounts(id) ON DELETE CASCADE
);

-- Indexes for performance
CREATE INDEX IF NOT EXISTS idx_sa_refresh_tokens_token ON service_account_refresh_tokens(token);
CREATE INDEX IF NOT EXISTS idx_sa_refresh_tokens_service_account_id ON service_account_refresh_tokens(service_account_id);

-- Comment for documentation
COMMENT ON TABLE service_account_refresh_tokens IS 'Stores refresh tokens for service account JWT token refresh mechanism';
